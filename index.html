from fastapi import FastAPI
from databases import Database
from redis import Redis

app = FastAPI()
db  = Database("postgresql://...")
rds = Redis(host="localhost")

@app.get("/api/users")
async def get_users():
    cached = rds.get("users")
    if cached:
        return cached
    return await db.fetch_all(
        "SELECT * FROM users"
    )
